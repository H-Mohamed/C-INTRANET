
CREATE TABLE `ccis_pvr`.`PVR` (`ID` INT PRIMARY KEY , `OBJET` VARCHAR(120) NOT NULL , `DATE` DATE NOT NULL , `HEURE` VARCHAR(120) NOT NULL , `LIEU` VARCHAR(120) NOT NULL , `DEPARTEMENT` VARCHAR(120) NOT NULL , `SERVICE` VARCHAR(120) NULL , `ChargeSuivi` VARCHAR(120) NOT NULL , `SCAN` VARCHAR(200) NOT NULL , `IDEVENEMENT` INT(120) NULL ) ENGINE = InnoDB;

CREATE TABLE `ccis_pvr`.`images` (`ID` INT PRIMARY KEY AUTO_INCREMENT, `IDPVR` INT , `SRC` DATE NOT NULL ) ENGINE = InnoDB;
